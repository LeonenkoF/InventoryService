CREATE TABLE department
(
	dep_id smallint PRIMARY KEY,
	dep_name varchar(128)
);

CREATE TABLE inventory
(
	fk_dep_id smallint REFERENCES department(dep_id),
	inventory_id smallint PRIMARY KEY,
	workplace_number varchar(16) NOT NULL,
	full_name varchar(64) NOT NULL,
	pc_id varchar(16) NOT NULL,
	inventory_num varchar(16) NOT NULL,
	invent_monitors text NOT NULL,
	invent_printer text,
	invent_MFU text,
	invent_laptop text,
	invent_other text
)
